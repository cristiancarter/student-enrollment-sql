Interactive SQL Tutorial: Building a Student Enrollment System

This project highlights the use of SQL in building a student enrollment database.

STEP 1: Setup Database


CREATE DATABASE UK_University;
USE UK_University;

Database 'UK_University' created and selected for use.

STEP 2: Create Tables


CREATE TABLE Students (
    StudentID INT PRIMARY KEY AUTO_INCREMENT,
    FirstName VARCHAR(50),
    LastName VARCHAR(50),
    DateOfBirth DATE,
    Email VARCHAR(100)
);

CREATE TABLE Courses (
    CourseID INT PRIMARY KEY AUTO_INCREMENT,
    CourseName VARCHAR(100),
    Department VARCHAR(50)
);

CREATE TABLE Enrolments (
    EnrolmentID INT PRIMARY KEY AUTO_INCREMENT,
    StudentID INT,
    CourseID INT,
    Grade CHAR(2),
    FOREIGN KEY (StudentID) REFERENCES Students(StudentID),
    FOREIGN KEY (CourseID) REFERENCES Courses(CourseID)
);

STEP 5: Update Grade


UPDATE Enrolments
SET Grade = 'A'
WHERE StudentID = 2 AND CourseID = 2;

STEP 6: Create View


CREATE VIEW StudentGrades AS
SELECT 
    s.FirstName,
    s.LastName,
    c.CourseName,
    e.Grade
FROM 
    Enrolments e
    JOIN Students s ON e.StudentID = s.StudentID
    JOIN Courses c ON e.CourseID = c.CourseID;

This view combines data from Students, Courses, and Enrolments tables.

STEP 7: Stored Procedures


DELIMITER //

CREATE PROCEDURE EnrolStudent (
    IN p_StudentID INT,
    IN p_CourseID INT
)
BEGIN
    INSERT INTO Enrolments (StudentID, CourseID, Grade)
    VALUES (p_StudentID, p_CourseID, 'NA');
END //

DELIMITER ;
                            
CALL EnrolStudent(4, 1);

This would enrol a student with StudentID 4 into the course with CourseID 1, with an initial grade of 'NA'.

Example:

STEP 8: CASE Statement


SELECT 
    StudentID,
    CourseID,
    Grade,
    CASE
        WHEN Grade = 'A' THEN 'Excellent'
        WHEN Grade = 'B' THEN 'Good'
        WHEN Grade = 'C' THEN 'Average'
        ELSE 'Needs Improvement'
    END AS Performance
FROM Enrolments;
